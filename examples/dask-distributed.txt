[noderef scheduler]

[cluster dask-distributed]
FormLayout = selectionpanel
Category = Schedulers
IconUrl = https://github.com/Azure/cyclecloud-conda/raw/master/examples/dask.png

Autoscale = $Autoscale

    [[node defaults]]
    UsePublicNetwork = $UsePublicNetwork
    Credentials = $Credentials    
    ImageName = $ImageName
    SubnetId = $SubnetId
    Region = $Region
    KeyPairLocation = ~/.ssh/cyclecloud.pem
    
        [[[configuration]]]
        anaconda.channels = $configuration_anaconda_channels
	anaconda.env_files = $configuration_anaconda_env_files
	anaconda.python = $configuration_anaconda_python
	anaconda.install_type = $configuration_anaconda_install_type
	anaconda.packages = $configuration_anaconda_packages

	# Do not start Dask services if running on top of a batch scheduler
	dask.start_services = true
	dask.scheduler.port = 8786
	dask.dashboard.port = 8787

        [[[cluster-init anaconda:default:1.0.2]]]
        [[[cluster-init anaconda:dask:1.0.2]]]



    [[node scheduler]]
    MachineType = $MasterMachineType
    IsReturnProxy = $ReturnProxy
    AdditionalClusterInitSpecs = $MasterClusterInitSpecs
    

        [[[configuration]]]
	dask.is_scheduler = true
	dask.slot_count = 1

        [[[network-interface eth0]]]
        AssociatePublicIpAddress = $UsePublicNetwork

        [[[input-endpoint ganglia]]]
        PrivatePort = 8652
        PublicPort = 8652

        [[[input-endpoint dask]]]
        PrivatePort = 8787
        PublicPort = 8787

        [[[input-endpoint daskdebug]]]
        PrivatePort = 8788
        PublicPort = 8788


    [[nodearray workers]]
    MachineType = $ExecuteMachineType
    MaxCoreCount = $MaxExecuteCoreCount

    Interruptible = $UseLowPrio
    AdditionalClusterInitSpecs = $ExecuteClusterInitSpecs

        [[[configuration]]]
	dask.scheduler_addr = ${scheduler.instance.PrivateIp}


        [[[network-interface eth0]]]
        AssociatePublicIpAddress = $ExecuteNodesPublic

        HideLabel = true

        [[[parameter ExecuteMachineType]]]

    [[parameters Auto-Scaling]]

        [[[parameter Autoscale]]]

        DefaultValue = true

    [[parameters Networking]]

        [[[parameter SubnetId]]]

    [[parameters Anaconda Settings ]]

	[[[parameter configuration_anaconda_env_files]]]

	[[[parameter configuration_anaconda_install_type]]]

        [[[parameter ImageName]]]
        Label = Base OS
        ParameterType = Cloud.Image
        Config.OS = linux

        [[[parameter MasterClusterInitSpecs]]]
        Label = Master Cluster-Init
        DefaultValue = =undefined

        [[[parameter ExecuteClusterInitSpecs]]]

        ParameterType = Cloud.ClusterInitSpecs

        [[[parameter UsePublicNetwork]]]
        Label = Public Head Node
        DefaultValue = true

        [[[parameter ExecuteNodesPublic]]]
        Label = Public Execute
        DefaultValue = false
        ParameterType = Boolean

        Conditions.Excluded := UsePublicNetwork isnt true
